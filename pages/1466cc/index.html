Linux安装Tesseract

# Linux 安装 Tesseract

# 下载需要的包

# tesseract 需要单独下载

https://github.com/tesseract-ocr/tesseract/releases/tag/5.3.3
1

# 下载leptonica

wget http://www.leptonica.org/source/leptonica-1.78.0.tar.gz
1

# 安装依赖

# 解压leptonica

tar -xvf leptonica-1.78.0.tar.gz
1

# 配置编译安装leptonica,进入到文件夹中

./configure
make
make install
1
2
3

# 安装automake和libtool

yum -y install automake libtool
1

# 配置leptonica环境变量

# 修改profile,可以使用命令行或者直接编辑

vim /etc/profile
1

# 在文件结尾添加

export LD_LIBRARY_PATH=/usr/local/lib
export LIBLEPT_HEADERSDIR=/usr/local/include
export PKG_CONFIG_PATH=/usr/local/lib/pkgconfig
1
2
3

# 保存并退出文件

# 执行以下命令使配置生效

source /etc/profile
1

# 安装tesseract-ocr

# 安装依赖

yum install -y centos-release-scl devtoolset-8-gcc*
1

# 切换当前会话中gcc版本为8

scl enable devtoolset-8 bash
1

# 安装

yum install gcc-c++ libstdc++-devel
tar -xvf tesseract-4.0.0.tar.gz
cd ./tesseract-4.0.0
./autogen.sh
./configure
make
make install
1
2
3
4
5
6
7

# 测试是否安装成功,执行ldconfig

tesseract --version
